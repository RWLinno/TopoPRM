\section{Method}
\label{sec:method}

We present TopoPRM, a topology-aware process reward framework for
reinforcement fine-tuning of mathematical reasoning models.
Our approach consists of four stages: supervised fine-tuning with
structured reasoning data (\S\ref{sec:sft}), DAG-based process reward
extraction (\S\ref{sec:topo_rewards}), composite RLVR with stratified
advantage estimation (\S\ref{sec:composite_rlvr}), and reasoning
compression via reverse-KL distillation (\S\ref{sec:compression}).
Figure~\ref{fig:overview} illustrates the complete pipeline.

\subsection{Supervised Fine-Tuning with Structured Reasoning}
\label{sec:sft}

Given a mathematical problem $q$ and, optionally, a student solution $s$,
the model $\pi_\theta$ generates a structured response
$c = \pi_\theta(q, s)$ consisting of a reasoning trace enclosed in
\texttt{<think>$\cdots$</think>} tags and a structured answer in
\texttt{<answer>$\cdots$</answer>} tags. The reasoning trace is a sequence
of reasoning steps $\{r_1, r_2, \ldots, r_T\}$, each implicitly depending
on earlier steps. These dependencies induce a directed acyclic graph (DAG)
$\mathcal{G} = (\mathcal{V}, \mathcal{E})$, where $\mathcal{V} = \{r_1,
\ldots, r_T\}$ and $(r_i, r_j) \in \mathcal{E}$ if step $r_j$ logically
depends on step $r_i$.

We first fine-tune $\pi_\theta$ on a curated dataset of mathematical
reasoning traces using LoRA~\citep{hu2022lora}, teaching the model the
desired output format and structured reasoning patterns. This SFT stage
provides a warm-start policy for subsequent reinforcement learning.

\subsection{Topology-Aware Verifiable Rewards}
\label{sec:topo_rewards}

We design two topology-aware reward components that evaluate the
\emph{structural quality} of the reasoning DAG extracted from each model
completion. Both rewards are computed by deterministic programs and thus
satisfy the verifiability requirement of RLVR.

\subsubsection{DAG Extraction}
\label{sec:dag_extraction}

Given a reasoning trace $c$, we extract the DAG $\mathcal{G} =
(\mathcal{V}, \mathcal{E})$ using a rule-based parser:
\begin{enumerate}[leftmargin=*,itemsep=1pt]
    \item \textbf{Step segmentation.} We identify step boundaries via
    structural markers (numbered steps, mathematical delimiters, logical
    connectives such as $\because$, $\therefore$, and key phrases).
    Each segment becomes a node $r_i \in \mathcal{V}$.
    \item \textbf{Dependency detection.} For each pair $(r_i, r_j)$ with
    $i < j$, we check whether $r_j$ references mathematical expressions,
    variables, or conclusions from $r_i$. Detected references yield edges
    $(r_i, r_j) \in \mathcal{E}$. Additionally, consecutive steps receive
    weak sequential edges with lower weight.
    \item \textbf{Step classification.} Each node is classified into one of
    seven types: \textsc{definition}, \textsc{derivation},
    \textsc{computation}, \textsc{conclusion}, \textsc{auxiliary},
    \textsc{substitution}, or \textsc{case\_analysis}, based on keyword
    patterns.
\end{enumerate}
This extraction is fully deterministic and introduces no learned components
into the reward computation pipeline.

\subsubsection{Topological Structure Reward}
\label{sec:r_topo}

The topological structure reward $R_{\mathrm{topo}}$ evaluates the
extracted DAG against four structural properties:
\begin{equation}
\label{eq:r_topo}
R_{\mathrm{topo}}(\mathcal{G}) = R_{\mathrm{base}} + R_{\mathrm{acyc}}
    + R_{\mathrm{conn}} + R_{\mathrm{dir}},
\end{equation}
where:
\begin{itemize}[leftmargin=*,itemsep=1pt]
    \item $R_{\mathrm{base}} = 0.4 \cdot \mathbb{1}[|\mathcal{V}| > 0]$
    is a base score for extracting any reasoning steps;
    \item $R_{\mathrm{acyc}} = 0.2 \cdot \mathbb{1}[\mathcal{G} \text{ is
    acyclic}]$ gates on the absence of circular reasoning;
    \item $R_{\mathrm{conn}} = 0.2 \cdot (1 - |\mathcal{V}_{\mathrm{orphan}}|
    / |\mathcal{V}|)$ penalises orphan conclusion nodes that lack
    supporting predecessors;
    \item $R_{\mathrm{dir}} = 0.1 \cdot \delta(\mathcal{G})$ rewards
    directional consistency, where $\delta(\mathcal{G}) \in [0, 1]$
    measures the fraction of edges that maintain a forward reasoning
    direction (no conclusion$\to$premise back-edges).
\end{itemize}
Optionally, when a reference DAG $\mathcal{G}^*$ is available, we add a
coverage term $R_{\mathrm{cov}} = 0.1 \cdot |\mathcal{E} \cap
\mathcal{E}^*| / |\mathcal{E}^*|$ measuring overlap with the reference
dependency structure. The total reward satisfies $R_{\mathrm{topo}} \in
[0, 1]$.

\subsubsection{Continuity Reward}
\label{sec:r_cont}

The continuity reward $R_{\mathrm{cont}}$ captures the step-to-step
semantic coherence of the reasoning trace:
\begin{equation}
\label{eq:r_cont}
R_{\mathrm{cont}}(c) = \frac{|\{i : \mathrm{supported}(r_i, r_{<i})\}|}{T},
\end{equation}
where $\mathrm{supported}(r_i, r_{<i})$ evaluates whether step $r_i$
references only mathematical expressions and conclusions that appear in
earlier steps $r_{<i}$ or in the problem statement. Steps that introduce
novel claims without prior support (``skipped steps'') decrease the
continuity score. This reward is computed via regex-based expression
matching and requires no neural model.

\subsection{Composite RLVR with Stratified Advantage Estimation}
\label{sec:composite_rlvr}

\subsubsection{Composite Reward}
\label{sec:composite_reward}

We compose the topology-aware rewards with outcome and format rewards:
\begin{equation}
\label{eq:r_total}
R_{\mathrm{total}} = w_o R_{\mathrm{outcome}} + w_t R_{\mathrm{topo}}
    + w_c R_{\mathrm{cont}} + w_f R_{\mathrm{format}}
    + w_l R_{\mathrm{len}},
\end{equation}
where $R_{\mathrm{outcome}} \in \{0, 1\}$ checks answer correctness,
$R_{\mathrm{format}} \in [0, 1]$ checks adherence to required output
structure (\texttt{<think>}/\texttt{<answer>} tags with valid JSON), and
$R_{\mathrm{len}} \in [0, 1]$ is a length penalty that linearly decays
for outputs exceeding a threshold. All five components are computed
programmatically. Default weights are $w_o = 0.4$, $w_t = 0.2$,
$w_c = 0.15$, $w_f = 0.15$, $w_l = 0.1$.

\subsubsection{Stratified Clipping Advantage Estimation}
\label{sec:scae}

A key challenge when combining process-level and outcome-level rewards
is \emph{reward hacking}: the model may learn to produce topologically
well-structured but incorrect reasoning. To address this, we propose
\textbf{Stratified Clipping Advantage Estimation (SCAE)}, which
hierarchically prioritises answer correctness over structural quality.

For each group of $G$ sampled completions, SCAE proceeds as follows:

\paragraph{Group stratification.}
We compute the mean accuracy reward $\bar{r}_{\mathrm{acc}}$ and
partition the group into a correct set
$\mathcal{C} = \{i : R_{\mathrm{outcome}}^{(i)} = 1\}$ and an incorrect
set $\mathcal{W} = \{j : R_{\mathrm{outcome}}^{(j)} = 0\}$. The
baseline accuracy advantages are:
\begin{equation}
\label{eq:a_acc}
A_{\mathrm{acc}}^{(i)} = \begin{cases}
    1 - \bar{r}_{\mathrm{acc}} & \text{if } i \in \mathcal{C}, \\
    -\bar{r}_{\mathrm{acc}} & \text{if } i \in \mathcal{W}.
\end{cases}
\end{equation}

\paragraph{Asymmetric auxiliary clipping.}
Within each stratum, we compute the mean auxiliary reward
$\bar{r}_{\mathrm{aux}}^{+}$ (for $\mathcal{C}$) and
$\bar{r}_{\mathrm{aux}}^{-}$ (for $\mathcal{W}$), where the auxiliary
reward aggregates all non-outcome components. We then apply asymmetric
clipping:
\begin{align}
A^{(i)} &= A_{\mathrm{acc}}^{(i)} + \max\!\big(0,\;
    r_{\mathrm{aux}}^{(i)} - \bar{r}_{\mathrm{aux}}^{+}\big),
    \quad i \in \mathcal{C}, \label{eq:a_correct} \\
A^{(j)} &= A_{\mathrm{acc}}^{(j)} + \min\!\big(0,\;
    r_{\mathrm{aux}}^{(j)} - \bar{r}_{\mathrm{aux}}^{-}\big),
    \quad j \in \mathcal{W}. \label{eq:a_wrong}
\end{align}

This ensures two critical properties:
\begin{itemize}[leftmargin=*,itemsep=1pt]
    \item \textbf{Accuracy primacy:} Correct completions always have
    $A^{(i)} \geq 1 - \bar{r}_{\mathrm{acc}} \geq 0$; incorrect
    completions always have $A^{(j)} \leq -\bar{r}_{\mathrm{acc}} \leq 0$.
    \item \textbf{Reward hacking resilience:} High structural quality on
    an incorrect trace provides no advantage bonus; low structural quality
    on a correct trace incurs no penalty.
\end{itemize}

\subsubsection{GRPO Optimisation}

We optimise $\pi_\theta$ using Group Relative Policy Optimisation
(GRPO)~\citep{shao2024deepseekmath} with the stratified advantages from
SCAE replacing the standard normalised advantage. The policy gradient loss
retains the standard PPO clipping and KL regularisation:
\begin{equation}
\label{eq:grpo_loss}
\mathcal{L}_{\mathrm{GRPO}} = -\mathbb{E}\bigg[
    \min\!\Big(\rho_t \hat{A}^{(i)},\;
    \mathrm{clip}(\rho_t, 1\!-\!\epsilon, 1\!+\!\epsilon)\hat{A}^{(i)}
    \Big) - \beta\, D_{\mathrm{KL}}(\pi_\theta \| \pi_{\mathrm{ref}})
\bigg],
\end{equation}
where $\rho_t = \pi_\theta(a_t | s_t) / \pi_{\mathrm{ref}}(a_t | s_t)$
is the importance ratio, $\epsilon$ is the clipping parameter, and $\beta$
controls the KL penalty.

\subsection{Reasoning Compression via Reverse-KL Distillation}
\label{sec:compression}

After GRPO training, the 32B teacher $\pi_{\theta^*}$ produces accurate
but sometimes verbose reasoning traces. We distil the teacher's reasoning
behaviour into smaller student models $\pi_\phi$ by minimising the reverse
KL divergence:
\begin{equation}
\label{eq:distill}
\mathcal{L}_{\mathrm{distill}} = \mathbb{E}_{(q,s) \sim \mathcal{D}}
    \Big[ D_{\mathrm{KL}}\!\big(\pi_\phi(\cdot | q, s)
    \,\|\, \pi_{\theta^*}(\cdot | q, s)\big) \Big].
\end{equation}
Reverse KL (mode-seeking) encourages the student to concentrate
probability mass on the teacher's highest-quality outputs rather than
covering all plausible completions. The distillation dataset consists of
teacher-generated traces filtered to retain only those with
$R_{\mathrm{total}} > \tau$ for a quality threshold $\tau$.

\begin{algorithm}[t]
\caption{TopoPRM Training Pipeline}
\label{alg:topoprm}
\begin{algorithmic}[1]
\REQUIRE Base model $\pi_0$, SFT data $\mathcal{D}_{\mathrm{sft}}$, GRPO prompts $\mathcal{D}_{\mathrm{grpo}}$, reward weights $\{w_o, w_t, w_c, w_f, w_l\}$
\ENSURE Trained policy $\pi_{\theta^*}$

\STATE \textbf{Stage 1: SFT} \\
Train $\pi_\theta \leftarrow \mathrm{LoRA\text{-}SFT}(\pi_0, \mathcal{D}_{\mathrm{sft}})$

\STATE \textbf{Stage 2: GRPO with TopoPRM}
\FOR{each prompt batch $\{q_i\}$ from $\mathcal{D}_{\mathrm{grpo}}$}
    \STATE Sample $G$ completions: $\{c^{(g)}\}_{g=1}^G \sim \pi_\theta(\cdot | q_i)$
    \STATE Compute $R_{\mathrm{outcome}}^{(g)}, R_{\mathrm{topo}}^{(g)}, R_{\mathrm{cont}}^{(g)}, R_{\mathrm{format}}^{(g)}, R_{\mathrm{len}}^{(g)}$ for each $g$
    \STATE $R_{\mathrm{total}}^{(g)} = w_o R_o^{(g)} + w_t R_t^{(g)} + w_c R_c^{(g)} + w_f R_f^{(g)} + w_l R_l^{(g)}$
    \STATE Compute SCAE advantages $\hat{A}^{(g)}$ via Eqs.~\ref{eq:a_acc}--\ref{eq:a_wrong}
    \STATE Update $\pi_\theta$ via clipped GRPO objective (Eq.~\ref{eq:grpo_loss})
\ENDFOR

\STATE \textbf{Stage 3: Distillation} (optional) \\
Generate filtered traces $\{(q, c^*) : R_{\mathrm{total}}(c^*) > \tau\}$ \\
Train student $\pi_\phi$ via reverse-KL distillation (Eq.~\ref{eq:distill})

\RETURN $\pi_{\theta^*}$ (or $\pi_\phi$ for compressed model)
\end{algorithmic}
\end{algorithm}
